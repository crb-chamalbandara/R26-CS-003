import json
import os

# Collect all malicious Extension IDs from your cloned repos
malicious_ids = set()

# From chrome-mal-ids repo
# Change this path to where you cloned the repo
mal_ids_path = "C:/path/to/chrome-mal-ids"
for filename in os.listdir(mal_ids_path):
    if filename.endswith(".txt"):
        with open(os.path.join(mal_ids_path, filename)) as f:
            for line in f:
                ext_id = line.strip()
                if ext_id and len(ext_id) == 32:
                    malicious_ids.add(ext_id)

# From palant malicious-extensions-list repo
# Check the actual file format — could be JSON, CSV, or txt
palant_path = "C:/path/to/malicious-extensions-list"
for filename in os.listdir(palant_path):
    if filename.endswith(".json"):
        with open(os.path.join(palant_path, filename)) as f:
            data = json.load(f)
            if isinstance(data, list):
                for item in data:
                    if isinstance(item, dict) and "id" in item:
                        malicious_ids.add(item["id"])
                    elif isinstance(item, str) and len(item) == 32:
                        malicious_ids.add(item)

# Save to JSON file
output = {"malicious_extension_ids": list(malicious_ids)}
with open("malicious_ids.json", "w") as f:
    json.dump(output, f, indent=2)

print(f"✓ Saved {len(malicious_ids)} malicious extension IDs to malicious_ids.json")

backend/static_analysis.pyPython
import hashlib
import zipfile
import json
import os
import joblib
import pandas as pd
import numpy as np

# ── Load resources on startup ──────────────────────────────────
BASE_DIR = os.path.dirname(os.path.abspath(__file__))
MODEL_PATH = os.path.join(BASE_DIR, "../ml_model/extension_detector_model.pkl")
HASH_DB_PATH = os.path.join(BASE_DIR, "malicious_ids.json")

# Load the trained XGBoost model
model = joblib.load(MODEL_PATH)
print("✓ ML model loaded")

# Load malicious extension IDs
with open(HASH_DB_PATH) as f:
    hash_data = json.load(f)
MALICIOUS_IDS = set(hash_data.get("malicious_extension_ids", []))
print(f"✓ Hash database loaded: {len(MALICIOUS_IDS)} known malicious IDs")

# The 13 most important permission features
PERMISSION_FEATURES = [
    "has_webRequest", "has_all_urls", "has_cookies",
    "has_clipboardRead", "has_nativeMessaging", "has_tabs",
    "has_history", "has_downloads", "has_storage",
    "has_background_script", "has_content_scripts",
    "host_permission_count", "total_permission_count"
]

# ── FUNCTION 1: Compute SHA-256 hash of CRX file ──────────────
def compute_file_hash(file_path):
    sha256 = hashlib.sha256()
    with open(file_path, "rb") as f:
        for chunk in iter(lambda: f.read(4096), b""):
            sha256.update(chunk)
    return sha256.hexdigest()

# ── FUNCTION 2: Extract Extension ID from CRX filename ────────
def extract_extension_id(filename):
    # Extension IDs are 32 lowercase letters
    name = os.path.splitext(os.path.basename(filename))[0]
    if len(name) == 32 and name.isalpha() and name.islower():
        return name
    return "unknown"

# ── FUNCTION 3: Parse manifest.json from CRX file ─────────────
def parse_manifest(crx_path):
    try:
        with zipfile.ZipFile(crx_path, "r") as zf:
            # CRX v3 files have extra header bytes — try direct zip first
            if "manifest.json" in zf.namelist():
                with zf.open("manifest.json") as mf:
                    return json.loads(mf.read().decode("utf-8", errors="ignore"))
    except zipfile.BadZipFile:
        # CRX3 format has a header to strip — try offset approach
        with open(crx_path, "rb") as f:
            data = f.read()
        # Find start of ZIP data (PK signature)
        zip_start = data.find(b"PK\x03\x04")
        if zip_start > 0:
            import io
            zip_data = io.BytesIO(data[zip_start:])
            with zipfile.ZipFile(zip_data) as zf:
                if "manifest.json" in zf.namelist():
                    with zf.open("manifest.json") as mf:
                        return json.loads(mf.read().decode("utf-8", errors="ignore"))
    return {}

# ── FUNCTION 4: Convert manifest to feature vector ────────────
def manifest_to_features(manifest):
    perms = manifest.get("permissions", [])
    host_perms = manifest.get("host_permissions", [])
    bg = manifest.get("background", {})

    # Convert all permissions to lowercase strings
    perm_set = set(str(p).lower() for p in perms)

    features = {
        "has_webRequest":       1 if "webrequest" in perm_set else 0,
        "has_all_urls":         1 if any("<all_urls>" in p or "*://*/*" in p for p in perms + host_perms) else 0,
        "has_cookies":          1 if "cookies" in perm_set else 0,
        "has_clipboardRead":    1 if "clipboardread" in perm_set else 0,
        "has_nativeMessaging":  1 if "nativemessaging" in perm_set else 0,
        "has_tabs":             1 if "tabs" in perm_set else 0,
        "has_history":          1 if "history" in perm_set else 0,
        "has_downloads":        1 if "downloads" in perm_set else 0,
        "has_storage":          1 if "storage" in perm_set else 0,
        "has_background_script":1 if bg else 0,
        "has_content_scripts":  1 if manifest.get("content_scripts") else 0,
        "host_permission_count": len(host_perms),
        "total_permission_count": len(perms)
    }
    return features

# ── FUNCTION 5: Main static analysis entry point ──────────────
def run_static_analysis(crx_path):
    """
    Runs complete static analysis on a CRX file.
    Returns a dict with all analysis results.
    """
    result = {
        "file_path": crx_path,
        "file_hash": "",
        "extension_id": "",
        "hash_match": False,
        "manifest": {},
        "features": {},
        "ml_score": 0,
        "ml_probability": 0.0,
        "risk_flags": [],
        "static_score": 0,
        "verdict": "SAFE"
    }

    # STEP 1: Hash check
    file_hash = compute_file_hash(crx_path)
    ext_id = extract_extension_id(crx_path)
    result["file_hash"] = file_hash
    result["extension_id"] = ext_id

    if ext_id in MALICIOUS_IDS:
        result["hash_match"] = True
        result["risk_flags"].append("HASH_MATCH: Extension ID found in malicious database")
        result["static_score"] = 100
        result["verdict"] = "MALICIOUS"
        return result  # No need to continue

    # STEP 2: Parse manifest
    manifest = parse_manifest(crx_path)
    result["manifest"] = manifest

    if not manifest:
        result["risk_flags"].append("PARSE_ERROR: Could not read manifest.json")
        result["static_score"] = 50
        result["verdict"] = "SUSPICIOUS"
        return result

    # STEP 3: Extract features and add risk flags
    features = manifest_to_features(manifest)
    result["features"] = features

    if features["has_webRequest"] and features["has_all_urls"]:
        result["risk_flags"].append("PERM_COMBO: webRequest + all_urls — can intercept all network requests")
    if features["has_clipboardRead"]:
        result["risk_flags"].append("PERM_CLIP: clipboardRead permission — can steal clipboard content")
    if features["has_nativeMessaging"]:
        result["risk_flags"].append("PERM_NATIVE: nativeMessaging — can communicate outside browser")
    if features["has_cookies"] and features["has_all_urls"]:
        result["risk_flags"].append("PERM_COOKIE: Cookies + all_urls — can steal session cookies from all sites")
    if features["total_permission_count"] > 10:
        result["risk_flags"].append(f"PERM_COUNT: {features['total_permission_count']} permissions declared — unusually high")

    # STEP 4: ML classification
    try:
        df = pd.read_csv("../dataset_clean.csv", nrows=1)
        label_col = df.columns[-1]
        all_feature_cols = [c for c in df.columns if c != label_col]

        # Build a row matching the full feature set
        row = {}
        for col in all_feature_cols:
            row[col] = features.get(col, 0)
        input_df = pd.DataFrame([row])
        proba = model.predict_proba(input_df)[0][1]
        ml_score = int(proba * 100)
    except Exception as e:
        # Fallback: use just the 13 permission features
        input_df = pd.DataFrame([features])
        proba = 0.3
        ml_score = 30

    result["ml_score"] = ml_score
    result["ml_probability"] = round(float(proba), 4)

    # STEP 5: Combine into final static_score
    flag_score = len(result["risk_flags"]) * 10
    static_score = min(100, int(ml_score * 0.7 + flag_score * 0.3))
    result["static_score"] = static_score

    if static_score >= 70:
        result["verdict"] = "MALICIOUS"
    elif static_score >= 40:
        result["verdict"] = "SUSPICIOUS"
    else:
        result["verdict"] = "SAFE"

    return result

const puppeteer = require('puppeteer');
const path = require('path');
const fs = require('fs');

const CRX_PATH = process.env.CRX_PATH || process.argv[2];
const MONITOR_DURATION = 120000; // 120 seconds
const BAIT_URLS = [
    'https://www.google.com',
    'https://www.github.com',
    'https://httpbin.org/get'
];

let behaviourLog = {
    outbound_requests: [],
    background_requests: [],
    chrome_api_calls: [],
    flags: [],
    start_time: Date.now()
};

async function runSandbox() {
    console.error('[SANDBOX] Starting analysis...');

    const browser = await puppeteer.launch({
        headless: 'new',
        args: [
            '--no-sandbox',
            '--disable-setuid-sandbox',
            '--disable-dev-shm-usage',
            CRX_PATH ? `--load-extension=${CRX_PATH}` : '',
            '--disable-extensions-except=' + (CRX_PATH || '')
        ].filter(Boolean),
        executablePath: process.env.PUPPETEER_EXECUTABLE_PATH || undefined
    });

    const page = await browser.newPage();

    // Monitor all outbound network requests
    await page.setRequestInterception(true);
    page.on('request', request => {
        const url = request.url();
        const method = request.method();
        const timestamp = Date.now() - behaviourLog.start_time;

        behaviourLog.outbound_requests.push({
            url, method, timestamp,
            resourceType: request.resourceType()
        });

        // Flag suspicious outbound calls
        if (method === 'POST' && !isTrustedDomain(url)) {
            behaviourLog.flags.push({
                type: 'OUTBOUND_POST',
                detail: `POST to unknown host: ${new URL(url).hostname}`,
                timestamp
            });
        }
        request.continue();
    });

    // Navigate to bait URLs to trigger extension behaviour
    for (const url of BAIT_URLS) {
        try {
            await page.goto(url, { waitUntil: 'networkidle2', timeout: 15000 });
            await page.waitForTimeout(3000);
        } catch(e) {
            console.error(`[SANDBOX] Navigation error: ${e.message}`);
        }
    }

    // Wait remaining time to catch timer-based beaconing
    const elapsed = Date.now() - behaviourLog.start_time;
    const remaining = MONITOR_DURATION - elapsed;
    if (remaining > 0) {
        await page.waitForTimeout(remaining);
    }

    // Analyse beacon patterns in captured requests
    analyseBeaconPatterns();

    await browser.close();

    // Output results as JSON to stdout (Python reads this)
    process.stdout.write(JSON.stringify(behaviourLog));
}

function isTrustedDomain(url) {
    const trusted = ['google.com','googleapis.com','gstatic.com',
                     'github.com','github.io','httpbin.org'];
    try {
        const hostname = new URL(url).hostname;
        return trusted.some(t => hostname.endsWith(t));
    } catch { return false; }
}

function analyseBeaconPatterns() {
    // Group requests by destination host
    const hostMap = {};
    for (const req of behaviourLog.outbound_requests) {
        try {
            const host = new URL(req.url).hostname;
            if (!hostMap[host]) hostMap[host] = [];
            hostMap[host].push(req.timestamp);
        } catch {}
    }

    // Check for periodic beaconing (low coefficient of variation)
    for (const [host, times] of Object.entries(hostMap)) {
        if (times.length >= 3 && !isTrustedDomain(`https://${host}`)) {
            const deltas = times.slice(1).map((t, i) => t - times[i]);
            const mean = deltas.reduce((a,b) => a+b) / deltas.length;
            const variance = deltas.reduce((s,d) => s + (d-mean)**2, 0) / deltas.length;
            const cv = Math.sqrt(variance) / mean;

            if (cv < 0.15) {
                behaviourLog.flags.push({
                    type: 'BEACON_PATTERN',
                    detail: `Periodic requests to ${host} — CV=${cv.toFixed(3)} (beaconing)`,
                    host, call_count: times.length
                });
            }
        }
    }
}

runSandbox().catch(err => {
    console.error('[SANDBOX ERROR]', err.message);
    process.stdout.write(JSON.stringify({ flags: [], error: err.message }));
});

import subprocess
import json
import os
import time

# ── FUNCTION: Run sandbox analysis via Docker ──────────────────
def run_dynamic_analysis(crx_path):
    """
    Runs the extension inside a Docker sandbox and returns
    the list of suspicious behaviour flags detected.
    """
    result = {
        "sandbox_ran": False,
        "duration_seconds": 0,
        "outbound_request_count": 0,
        "dynamic_flags": [],
        "dynamic_score": 0,
        "error": None
    }

    if not os.path.exists(crx_path):
        result["error"] = "CRX file not found"
        return result

    crx_abs = os.path.abspath(crx_path)
    crx_dir = os.path.dirname(crx_abs)
    crx_filename = os.path.basename(crx_abs)
    start_time = time.time()

    try:
        # Run Docker container
        # -v mounts the CRX file into the container
        # -e passes the path as an environment variable
        # --rm removes container after it finishes
        # --network=none disables real internet (safety)
        docker_cmd = [
            "docker", "run", "--rm",
            "--network=none",           # no real internet access
            "--memory=512m",             # limit memory
            "--cpus=1",                  # limit CPU
            f"-v", f"{crx_dir}:/crx_files",
            "-e", f"CRX_PATH=/crx_files/{crx_filename}",
            "component1-sandbox"
        ]

        proc = subprocess.run(
            docker_cmd,
            capture_output=True,
            text=True,
            timeout=150              # 150s timeout (sandbox runs for 120s)
        )

        result["duration_seconds"] = round(time.time() - start_time, 1)
        result["sandbox_ran"] = True

        # Parse the JSON output from the sandbox script
        if proc.stdout:
            sandbox_data = json.loads(proc.stdout)
            flags = sandbox_data.get("flags", [])
            requests = sandbox_data.get("outbound_requests", [])

            result["outbound_request_count"] = len(requests)
            result["dynamic_flags"] = flags

            # Score based on number and severity of flags
            score = 0
            for flag in flags:
                flag_type = flag.get("type", "")
                if flag_type == "BEACON_PATTERN":   score += 40
                elif flag_type == "OUTBOUND_POST":   score += 25
                elif flag_type == "DOM_SCRAPE":      score += 30
                elif flag_type == "COOKIE_ACCESS":   score += 35
                else:                               score += 15

            result["dynamic_score"] = min(100, score)

    except subprocess.TimeoutExpired:
        result["error"] = "Sandbox timed out after 150 seconds"
        result["dynamic_score"] = 50  # timeout is suspicious
    except Exception as e:
        result["error"] = str(e)

    return result

from database import save_analysis_result
from report_generator import generate_report

def compute_final_verdict(static_result, dynamic_result=None):
    """
    Combines static + dynamic scores into one final verdict.
    dynamic_result is None if sandbox was not triggered.
    """

    static_score  = static_result.get("static_score", 0)
    dynamic_score = dynamic_result.get("dynamic_score", 0) if dynamic_result else 0
    hash_match    = static_result.get("hash_match", False)
    static_flags  = static_result.get("risk_flags", [])
    dynamic_flags = dynamic_result.get("dynamic_flags", []) if dynamic_result else []
    all_flags     = static_flags + [f.get("detail", str(f)) for f in dynamic_flags]
    sandbox_ran   = dynamic_result.get("sandbox_ran", False) if dynamic_result else False

    # CASE 1: Definite hash match — always MALICIOUS
    if hash_match:
        final_score = 100
        verdict = "MALICIOUS"
        confidence = "HIGH"

    # CASE 2: Both static and dynamic scores available
    elif sandbox_ran:
        # Weight: 40% static, 60% dynamic (behaviour is stronger evidence)
        final_score = int(static_score * 0.4 + dynamic_score * 0.6)
        confidence = "HIGH"
        if final_score >= 70:
            verdict = "MALICIOUS"
        elif final_score >= 40:
            verdict = "SUSPICIOUS"
        else:
            verdict = "SAFE"

    # CASE 3: Static analysis only (safe extensions, no sandbox run)
    else:
        final_score = static_score
        confidence = "MEDIUM"
        if final_score >= 70:
            verdict = "MALICIOUS"
        elif final_score >= 40:
            verdict = "SUSPICIOUS"
        else:
            verdict = "SAFE"

    verdict_result = {
        "extension_id":    static_result.get("extension_id", "unknown"),
        "file_hash":       static_result.get("file_hash", ""),
        "static_score":    static_score,
        "dynamic_score":   dynamic_score,
        "final_score":     final_score,
        "verdict":         verdict,
        "confidence":      confidence,
        "sandbox_ran":     sandbox_ran,
        "alert_flags":     all_flags,
        "flag_count":      len(all_flags),
        "manifest":        static_result.get("manifest", {}),
        "features":        static_result.get("features", {}),
    }

    # Save to database
    analysis_id = save_analysis_result(verdict_result)
    verdict_result["analysis_id"] = analysis_id

    # Generate human-readable report
    report = generate_report(verdict_result)
    verdict_result["report"] = report

    return verdict_result

import sqlite3

DB_PATH = os.path.join(os.path.dirname(__file__), "../data/component1.db")
os.makedirs(os.path.dirname(DB_PATH), exist_ok=True)

def init_db():
    """Create tables if they don't exist."""
    conn = sqlite3.connect(DB_PATH)
    conn.execute("""
        CREATE TABLE IF NOT EXISTS analyses (
            id          INTEGER PRIMARY KEY AUTOINCREMENT,
            analysed_at TEXT    NOT NULL,
            extension_id TEXT,
            file_hash   TEXT,
            static_score INTEGER,
            dynamic_score INTEGER,
            final_score INTEGER,
            verdict     TEXT,
            confidence  TEXT,
            sandbox_ran INTEGER,
            flag_count  INTEGER,
            alert_flags TEXT,
            manifest    TEXT,
            features    TEXT
        )
    """)
    conn.commit()
    conn.close()

init_db()

def save_analysis_result(result):
    """Save a verdict result to the database. Returns the new row ID."""
    conn = sqlite3.connect(DB_PATH)
    cursor = conn.execute("""
        INSERT INTO analyses
        (analysed_at, extension_id, file_hash, static_score,
         dynamic_score, final_score, verdict, confidence,
         sandbox_ran, flag_count, alert_flags, manifest, features)
        VALUES (?,?,?,?,?,?,?,?,?,?,?,?,?)
    """, (
        datetime.now().isoformat(),
        result.get("extension_id", ""),
        result.get("file_hash", ""),
        result.get("static_score", 0),
        result.get("dynamic_score", 0),
        result.get("final_score", 0),
        result.get("verdict", "UNKNOWN"),
        result.get("confidence", "LOW"),
        1 if result.get("sandbox_ran") else 0,
        result.get("flag_count", 0),
        json.dumps(result.get("alert_flags", [])),
        json.dumps(result.get("manifest", {})),
        json.dumps(result.get("features", {}))
    ))
    conn.commit()
    analysis_id = cursor.lastrowid
    conn.close()
    return analysis_id

def get_all_analyses():
    """Return all analyses for the dashboard."""
    conn = sqlite3.connect(DB_PATH)
    conn.row_factory = sqlite3.Row
    rows = conn.execute("SELECT * FROM analyses ORDER BY id DESC").fetchall()
    conn.close()
    result = []
    for row in rows:
        item = dict(row)
        item["alert_flags"] = json.loads(item["alert_flags"] or "[]")
        result.append(item)
    return result

def get_analysis_by_id(analysis_id):
    """Return one analysis by its ID."""
    conn = sqlite3.connect(DB_PATH)
    conn.row_factory = sqlite3.Row
    row = conn.execute("SELECT * FROM analyses WHERE id=?", (analysis_id,)).fetchone()
    conn.close()
    if not row: return None
    item = dict(row)
    item["alert_flags"] = json.loads(item["alert_flags"] or "[]")
    item["manifest"]    = json.loads(item["manifest"]    or "{}")
    item["features"]    = json.loads(item["features"]    or "{}")
    return item

from datetime import datetime

def generate_report(verdict):
    score   = verdict.get("final_score", 0)
    verdict_str = verdict.get("verdict", "UNKNOWN")
    flags   = verdict.get("alert_flags", [])
    sandbox = verdict.get("sandbox_ran", False)
    ext_id  = verdict.get("extension_id", "unknown")
    feats   = verdict.get("features", {})

    risk_bar = "█" * (score // 10) + "░" * (10 - score // 10)
    lines = [
        f"╔══════════════════════════════════════════════════╗",
        f"║   COMPONENT 1 — MALICIOUS EXTENSION ANALYZER     ║",
        f"║   Browser Security Intelligence · R26-CS-003      ║",
        f"╚══════════════════════════════════════════════════╝",
        f"",
        f"Extension ID  : {ext_id}",
        f"Analysed At   : {datetime.now().strftime('%Y-%m-%d %H:%M:%S')}",
        f"Sandbox Run   : {'Yes' if sandbox else 'No (static only)'}",
        f"",
        f"THREAT SCORE  : {score}/100  [{risk_bar}]",
        f"VERDICT       : *** {verdict_str} ***",
        f"",
        f"── Declared Permissions ──────────────────────────",
    ]

    perm_map = {
        "has_webRequest":       "webRequest",
        "has_all_urls":         "<all_urls> (all websites)",
        "has_cookies":          "cookies",
        "has_clipboardRead":    "clipboardRead ⚠",
        "has_nativeMessaging":  "nativeMessaging ⚠",
        "has_tabs":             "tabs",
        "has_downloads":        "downloads",
        "has_history":          "history",
        "has_background_script":"background script",
    }
    for key, label in perm_map.items():
        if feats.get(key):
            lines.append(f"  [✓] {label}")
    total = feats.get("total_permission_count", 0)
    lines.append(f"  Total: {total} permissions declared")

    if flags:
        lines += [f"", f"── Alert Flags ({len(flags)}) ─────────────────────────"]
        for i, flag in enumerate(flags, 1):
            lines.append(f"  {i}. {flag}")

    lines += [
        f"",
        f"── Recommendation ────────────────────────────────",
        f"  {'BLOCK — Do not install this extension.' if verdict_str == 'MALICIOUS' else 'CAUTION — Review manually before installing.' if verdict_str == 'SUSPICIOUS' else 'ALLOW — Extension appears safe to install.'}",
        f"═══════════════════════════════════════════════════"
    ]
    return "\n".join(lines)

from flask import Flask, request, jsonify
from flask_cors import CORS
import os

from static_analysis  import run_static_analysis
from dynamic_analysis import run_dynamic_analysis
from verdict_engine   import compute_final_verdict
from database         import get_all_analyses, get_analysis_by_id

app = Flask(__name__)
CORS(app)  # Allow frontend to call this API

UPLOAD_FOLDER = os.path.join(os.path.dirname(__file__), "../data/uploads")
os.makedirs(UPLOAD_FOLDER, exist_ok=True)

# ── ENDPOINT 1: Health check ───────────────────────────────────
@app.route("/api/health")
def health():
    return jsonify({"status": "ok", "message": "Component 1 API running"})

# ── ENDPOINT 2: Analyse an extension ──────────────────────────
@app.route("/api/analyse", methods=["POST"])
def analyse():
    if "file" not in request.files:
        return jsonify({"error": "No file uploaded"}), 400

    file = request.files["file"]
    if not file.filename.endswith(".crx"):
        return jsonify({"error": "Only .crx files accepted"}), 400

    # Save the uploaded file
    save_path = os.path.join(UPLOAD_FOLDER, file.filename)
    file.save(save_path)

    # MODULE 1: Static analysis
    static_result = run_static_analysis(save_path)

    # MODULE 2: Dynamic analysis (only if suspicious)
    dynamic_result = None
    run_sandbox = request.form.get("run_sandbox", "auto")
    if run_sandbox == "always" or (run_sandbox == "auto" and static_result["static_score"] >= 40):
        dynamic_result = run_dynamic_analysis(save_path)

    # MODULE 3: Verdict engine
    final = compute_final_verdict(static_result, dynamic_result)

    return jsonify(final)

# ── ENDPOINT 3: Get all past analyses ─────────────────────────
@app.route("/api/analyses")
def get_analyses():
    return jsonify(get_all_analyses())

# ── ENDPOINT 4: Get one analysis by ID ────────────────────────
@app.route("/api/analyses/<int:analysis_id>")
def get_analysis(analysis_id):
    result = get_analysis_by_id(analysis_id)
    if not result:
        return jsonify({"error": "Not found"}), 404
    return jsonify(result)

if __name__ == "__main__":
    print("Starting Component 1 API server...")
    app.run(debug=True, host="0.0.0.0", port=5001)

  setTimeout(() => {
    document.getElementById('loadingText').textContent = 'Running ML classifier...';

  setTimeout(() => {
    document.getElementById('loadingText').textContent = 'Sandbox analysis in progress (120s)...';
  }, 5000);

  try {
    const res = await fetch(`${API}/analyse`, { method: 'POST', body: formData });

  document.getElementById('scoreCircle').style.strokeDashoffset = 163 - (163 * score / 100);

Testing

Quick test checklist

✓
Flask backend starts without errors (python app.py runs and shows "Running on http://...")
✓
Health check works — open browser and go to http://localhost:5001/api/health — you should see {"status":"ok"}
✓
Frontend index.html loads in browser and history section shows (even if empty)
✓
Upload a .crx file — static analysis runs and returns a score and verdict
✓
Docker Desktop is open — rebuild Docker image if needed: docker build -t component1-sandbox . (run from sandbox folder)
✓
Sandbox analysis runs for a suspicious extension (try sandbox mode = "always")
✓
Analysis appears in the history section after completion
✓
Full Report tab shows the formatted text report with correct extension ID and verdict
If you get an import error when starting app.py
Make sure you are running the command from inside the backend/ folder, not the root component1_ml folder. All Python files import each other using relative imports which only work correctly from the backend directory.
